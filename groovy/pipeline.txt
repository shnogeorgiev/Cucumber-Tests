pipeline {
    agent any
    
    environment {
        CUCUMBER_FEATURES = 'src/test/java/features'
        CUCUMBER_GLUE = 'src/test/java/steps,src/test/java/utilities'
    }
    
    tools {
        maven 'Maven 3'
        jdk 'OpenJDK 11'
    }

    stages {
        stage('Checkout') {
            steps {
                // Checkout the repository containing the tests
                checkout scm
            }
        }

        stage('Install Dependencies') {
            steps {
                script {
                    bat 'mvn clean verify' // Runs compile & test but skips deploy/install
                }
            }
        }

        stage('Run Cucumber Tests') {
            steps {
                script {
                    bat 'mvn test -Dcucumber.options="--plugin json:target/cucumber.json"'
                }
            }
        }

        stage('Publish Cucumber Reports') {
            steps {
                cucumber(
                    fileIncludePattern: '**/target/cucumber.json',
                    jsonReportDirectory: 'target',
                    reportTitle: 'Cucumber Test Report'
                )
            }
        }
    }

    post {
        always {
            node {
                // Clean up, if needed (e.g., delete temp files)
                cleanWs()
            }
        }
    }
}
